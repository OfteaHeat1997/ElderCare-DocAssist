openapi: 3.0.3
info:
  title: VoiceVerzorging API
  description: |
    API for the VoiceVerzorging MVP - a healthcare documentation assistant that converts 
    speech to structured clinical notes for elderly care professionals.
  version: 1.0.0
  contact:
    name: VoiceVerzorging Team
    email: team@voiceverzorging.nl

servers:
  - url: https://api.voiceverzorging.nl/v1
    description: Production server
  - url: https://staging-api.voiceverzorging.nl/v1
    description: Staging server
  - url: http://localhost:3000/v1
    description: Local development server

tags:
  - name: patients
    description: Patient management operations
  - name: transcription
    description: Audio recording and transcription operations
  - name: documentation
    description: Clinical documentation operations

paths:
  /patients:
    get:
      tags:
        - patients
      summary: Get list of patients
      description: Retrieve a list of all patients assigned to the healthcare worker
      operationId: getPatients
      parameters:
        - name: limit
          in: query
          description: Maximum number of patients to return
          schema:
            type: integer
            default: 50
            minimum: 1
            maximum: 100
        - name: offset
          in: query
          description: Number of patients to skip
          schema:
            type: integer
            default: 0
            minimum: 0
      responses:
        '200':
          description: Successful response
          content:
            application/json:
              schema:
                type: object
                properties:
                  data:
                    type: array
                    items:
                      $ref: '#/components/schemas/Patient'
                  total:
                    type: integer
                    description: Total number of patients
                  limit:
                    type: integer
                  offset:
                    type: integer
        '401':
          $ref: '#/components/responses/UnauthorizedError'
        '500':
          $ref: '#/components/responses/InternalServerError'

  /patients/{patientId}:
    get:
      tags:
        - patients
      summary: Get patient details
      description: Retrieve detailed information about a specific patient
      operationId: getPatientById
      parameters:
        - name: patientId
          in: path
          required: true
          description: Unique identifier of the patient
          schema:
            type: string
            format: uuid
      responses:
        '200':
          description: Successful response
          content:
            application/json:
              schema:
                $ref: '#/components/schemas/Patient'
        '404':
          $ref: '#/components/responses/NotFoundError'
        '401':
          $ref: '#/components/responses/UnauthorizedError'

  /transcription/start:
    post:
      tags:
        - transcription
      summary: Start audio recording session
      description: Initialize a new audio recording and transcription session for a patient visit
      operationId: startRecording
      requestBody:
        required: true
        content:
          application/json:
            schema:
              type: object
              required:
                - patientId
              properties:
                patientId:
                  type: string
                  format: uuid
                  description: ID of the patient for this recording
                visitType:
                  type: string
                  enum: [medication, wound_care, daily_activities, therapy, general]
                  description: Type of care visit being documented
      responses:
        '201':
          description: Recording session started successfully
          content:
            application/json:
              schema:
                type: object
                properties:
                  sessionId:
                    type: string
                    format: uuid
                    description: Unique identifier for this recording session
                  patientId:
                    type: string
                    format: uuid
                  startedAt:
                    type: string
                    format: date-time
                  status:
                    type: string
                    enum: [recording]
        '400':
          $ref: '#/components/responses/BadRequestError'
        '401':
          $ref: '#/components/responses/UnauthorizedError'

  /transcription/{sessionId}/audio:
    post:
      tags:
        - transcription
      summary: Upload audio chunk
      description: Upload audio data for real-time or batch transcription
      operationId: uploadAudioChunk
      parameters:
        - name: sessionId
          in: path
          required: true
          description: Recording session identifier
          schema:
            type: string
            format: uuid
      requestBody:
        required: true
        content:
          audio/wav:
            schema:
              type: string
              format: binary
          audio/webm:
            schema:
              type: string
              format: binary
          audio/mp3:
            schema:
              type: string
              format: binary
      responses:
        '200':
          description: Audio chunk processed successfully
          content:
            application/json:
              schema:
                type: object
                properties:
                  transcription:
                    type: string
                    description: Partial or complete transcription text
                  confidence:
                    type: number
                    format: float
                    minimum: 0
                    maximum: 1
                    description: Confidence score of the transcription
        '400':
          $ref: '#/components/responses/BadRequestError'
        '404':
          $ref: '#/components/responses/NotFoundError'

  /transcription/{sessionId}/complete:
    post:
      tags:
        - transcription
      summary: Complete recording session
      description: Finalize the recording and get the complete transcription
      operationId: completeRecording
      parameters:
        - name: sessionId
          in: path
          required: true
          schema:
            type: string
            format: uuid
      responses:
        '200':
          description: Recording completed and transcription ready
          content:
            application/json:
              schema:
                $ref: '#/components/schemas/Transcription'
        '404':
          $ref: '#/components/responses/NotFoundError'

  /documentation/generate:
    post:
      tags:
        - documentation
      summary: Generate structured documentation
      description: Generate structured clinical documentation from transcription text
      operationId: generateDocumentation
      requestBody:
        required: true
        content:
          application/json:
            schema:
              type: object
              required:
                - transcriptionId
              properties:
                transcriptionId:
                  type: string
                  format: uuid
                  description: ID of the transcription to process
                documentationType:
                  type: string
                  enum: [standard, medication, incident, handover]
                  default: standard
      responses:
        '200':
          description: Documentation generated successfully
          content:
            application/json:
              schema:
                $ref: '#/components/schemas/ClinicalNote'
        '400':
          $ref: '#/components/responses/BadRequestError'
        '404':
          $ref: '#/components/responses/NotFoundError'

  /documentation/{noteId}:
    get:
      tags:
        - documentation
      summary: Get clinical note
      description: Retrieve a specific clinical note by ID
      operationId: getClinicalNote
      parameters:
        - name: noteId
          in: path
          required: true
          schema:
            type: string
            format: uuid
      responses:
        '200':
          description: Successful response
          content:
            application/json:
              schema:
                $ref: '#/components/schemas/ClinicalNote'
        '404':
          $ref: '#/components/responses/NotFoundError'
    
    patch:
      tags:
        - documentation
      summary: Update clinical note
      description: Update or correct a clinical note (human-in-the-loop review)
      operationId: updateClinicalNote
      parameters:
        - name: noteId
          in: path
          required: true
          schema:
            type: string
            format: uuid
      requestBody:
        required: true
        content:
          application/json:
            schema:
              $ref: '#/components/schemas/ClinicalNoteUpdate'
      responses:
        '200':
          description: Note updated successfully
          content:
            application/json:
              schema:
                $ref: '#/components/schemas/ClinicalNote'
        '400':
          $ref: '#/components/responses/BadRequestError'
        '404':
          $ref: '#/components/responses/NotFoundError'

  /documentation/{noteId}/approve:
    post:
      tags:
        - documentation
      summary: Approve clinical note
      description: Approve the clinical note for final submission
      operationId: approveClinicalNote
      parameters:
        - name: noteId
          in: path
          required: true
          schema:
            type: string
            format: uuid
      responses:
        '200':
          description: Note approved successfully
          content:
            application/json:
              schema:
                type: object
                properties:
                  noteId:
                    type: string
                    format: uuid
                  status:
                    type: string
                    enum: [approved]
                  approvedAt:
                    type: string
                    format: date-time
                  approvedBy:
                    type: string
                    description: Healthcare worker who approved the note
        '404':
          $ref: '#/components/responses/NotFoundError'

components:
  schemas:
    Patient:
      type: object
      properties:
        id:
          type: string
          format: uuid
          description: Unique patient identifier
        firstName:
          type: string
          example: Maria
        lastName:
          type: string
          example: Peters
        dateOfBirth:
          type: string
          format: date
          example: 1943-05-12
        roomNumber:
          type: string
          example: A204
          nullable: true
        careLevel:
          type: integer
          minimum: 1
          maximum: 5
          description: Care intensity level (1-5)
        primaryDiagnosis:
          type: string
          example: Dementia, Type 2 Diabetes
        allergies:
          type: array
          items:
            type: string
          example: [Penicillin, Peanuts]
        lastVisit:
          type: string
          format: date-time
          nullable: true

    Transcription:
      type: object
      properties:
        id:
          type: string
          format: uuid
        sessionId:
          type: string
          format: uuid
        patientId:
          type: string
          format: uuid
        transcriptionText:
          type: string
          description: Full transcription of the recorded audio
        confidence:
          type: number
          format: float
          minimum: 0
          maximum: 1
        duration:
          type: integer
          description: Recording duration in seconds
        createdAt:
          type: string
          format: date-time
        status:
          type: string
          enum: [processing, completed, failed]

    ClinicalNote:
      type: object
      properties:
        id:
          type: string
          format: uuid
        patientId:
          type: string
          format: uuid
        transcriptionId:
          type: string
          format: uuid
        noteType:
          type: string
          enum: [standard, medication, incident, handover]
        structuredData:
          type: object
          properties:
            symptoms:
              type: array
              items:
                type: string
              example: [Normal blood pressure 130/80]
            medications:
              type: array
              items:
                type: object
                properties:
                  name:
                    type: string
                  dosage:
                    type: string
                  administered:
                    type: boolean
              example: 
                - name: Metformin
                  dosage: 500mg
                  administered: true
            observations:
              type: string
              example: Patient was alert and cooperative
            interventions:
              type: string
              example: Administered morning medication via dispenser
            response:
              type: string
              example: Patient understood medication instructions
            nextSteps:
              type: string
              example: Next visit scheduled for Friday
        rawTranscription:
          type: string
          description: Original transcription text
        createdAt:
          type: string
          format: date-time
        updatedAt:
          type: string
          format: date-time
        createdBy:
          type: string
          description: Healthcare worker who created the note
        status:
          type: string
          enum: [draft, pending_review, approved]

    ClinicalNoteUpdate:
      type: object
      properties:
        structuredData:
          type: object
          properties:
            symptoms:
              type: array
              items:
                type: string
            medications:
              type: array
              items:
                type: object
            observations:
              type: string
            interventions:
              type: string
            response:
              type: string
            nextSteps:
              type: string

    Error:
      type: object
      properties:
        code:
          type: string
          description: Error code
        message:
          type: string
          description: Human-readable error message
        details:
          type: object
          additionalProperties: true
          description: Additional error details
        timestamp:
          type: string
          format: date-time

  responses:
    BadRequestError:
      description: Invalid request
      content:
        application/json:
          schema:
            $ref: '#/components/schemas/Error'
          example:
            code: BAD_REQUEST
            message: Invalid patient ID format
            timestamp: 2025-11-18T10:30:00Z

    UnauthorizedError:
      description: Authentication required
      content:
        application/json:
          schema:
            $ref: '#/components/schemas/Error'
          example:
            code: UNAUTHORIZED
            message: Valid authentication token required
            timestamp: 2025-11-18T10:30:00Z

    NotFoundError:
      description: Resource not found
      content:
        application/json:
          schema:
            $ref: '#/components/schemas/Error'
          example:
            code: NOT_FOUND
            message: Patient not found
            timestamp: 2025-11-18T10:30:00Z

    InternalServerError:
      description: Internal server error
      content:
        application/json:
          schema:
            $ref: '#/components/schemas/Error'
          example:
            code: INTERNAL_ERROR
            message: An unexpected error occurred
            timestamp: 2025-11-18T10:30:00Z

  securitySchemes:
    bearerAuth:
      type: http
      scheme: bearer
      bearerFormat: JWT
      description: JWT authentication token

security:
  - bearerAuth: []
